cmake_minimum_required(VERSION 2.8)

project(pdcex)


include_directories(source)

set(LIBS_PATH "${CMAKE_CURRENT_SOURCE_DIR}/libs")

include_directories("${LIBS_PATH}/include")
include_directories("${LIBS_PATH}/include/pdcurses")

IF(UNIX)
    set(CMAKE_CXX_FLAGS "-std=c++11")

    link_directories("${LIBS_PATH}/unix/pdcurses/")

    add_executable (pdce0 source/pdce0.c)
    add_executable (pdce1 source/pdce1.c)
    add_executable (pdce2 source/pdce2.c)
    add_executable (pdce3 source/pdce3.c)
    add_executable (pdce4 source/pdce4.c)
    add_executable (pdce5 source/pdce5.c)
    add_executable (pdce6 source/pdce6.c)
    add_executable (pdce7 source/pdce7.c)
    add_executable (pdce8 source/pdce8.c)
    add_executable (pdce9 source/pdce9.c)

    target_link_libraries(pdce0 m ncurses)
    target_link_libraries(pdce1 m ncurses)
    target_link_libraries(pdce2 m ncurses)
    target_link_libraries(pdce3 m ncurses)
    target_link_libraries(pdce4 m ncurses)
    target_link_libraries(pdce5 m ncurses)
    target_link_libraries(pdce6 m ncurses)
    target_link_libraries(pdce7 m ncurses)
    target_link_libraries(pdce8 m ncurses)
    target_link_libraries(pdce9 m ncurses)

ELSE (WIN32)
    link_directories("${LIBS_PATH}/windows/pdcurses/")

    add_executable (pdce0 source/pdce0.c)
    add_executable (pdce1 source/pdce1.c)
    add_executable (pdce2 source/pdce2.c)
    add_executable (pdce3 source/pdce3.c)
    add_executable (pdce4 source/pdce4.c)
    add_executable (pdce5 source/pdce5.c)
    add_executable (pdce6 source/pdce6.c)
    add_executable (pdce7 source/pdce7.c)
    add_executable (pdce8 source/pdce8.c)
    add_executable (pdce9 source/pdce9.c)

    target_link_libraries(pdce0 pdcurses.lib)
    target_link_libraries(pdce1 pdcurses.lib)
    target_link_libraries(pdce2 pdcurses.lib)
    target_link_libraries(pdce3 pdcurses.lib)
    target_link_libraries(pdce4 pdcurses.lib)
    target_link_libraries(pdce5 pdcurses.lib)
    target_link_libraries(pdce6 pdcurses.lib)
    target_link_libraries(pdce7 pdcurses.lib)
    target_link_libraries(pdce8 pdcurses.lib)
    target_link_libraries(pdce9 pdcurses.lib)

ENDIF (UNIX)

set(CMAKE_BUILD_TYPE Debug)
